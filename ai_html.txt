    <script>
      let scrapedData = null;
      let currentMode = 'ai';
      let aiSuggestedSelectors = null;

      function switchMode(mode) {
        currentMode = mode;
        const aiMode = document.getElementById('ai-mode');
        const manualMode = document.getElementById('manual-mode');
        const modeBtns = document.querySelectorAll('.mode-btn');

        modeBtns.forEach(btn => btn.classList.remove('active'));
        
        if (mode === 'ai') {
          aiMode.classList.remove('hidden');
          manualMode.classList.add('hidden');
          modeBtns[0].classList.add('active');
        } else {
          aiMode.classList.add('hidden');
          manualMode.classList.remove('hidden');
          modeBtns[1].classList.add('active');
        }
      }

      function addAIField() {
        const container = document.getElementById('ai-fields');
        const row = document.createElement('div');
        row.className = 'field-input-row';
        row.innerHTML = `
          <input type="text" placeholder="e.g., price" class="ai-field-input" />
          <button class="btn btn-danger btn-small" onclick="removeAIField(this)">‚úï</button>
        `;
        container.appendChild(row);
      }

      function removeAIField(btn) {
        btn.closest('.field-input-row').remove();
      }

      function addSelector() {
        const container = document.getElementById('selectors');
        const row = document.createElement('div');
        row.className = 'selector-row';
        row.innerHTML = `
          <div><input type="text" placeholder="Field name" class="selector-key"></div>
          <div><input type="text" placeholder="CSS selector" class="selector-value"></div>
          <div><input type="text" placeholder="Attribute (optional)" class="selector-attribute"></div>
          <div><button class="btn btn-danger btn-small" onclick="removeSelector(this)">‚úï</button></div>
        `;
        container.appendChild(row);
      }

      function removeSelector(btn) {
        btn.closest('.selector-row').remove();
      }

      async function analyzeWithAI() {
        const url = document.getElementById('url').value;
        const fieldInputs = document.querySelectorAll('.ai-field-input');
        const fields = [];

        fieldInputs.forEach(input => {
          if (input.value.trim()) {
            fields.push(input.value.trim());
          }
        });

        if (!url) {
          showError('Please enter a URL');
          return;
        }

        if (fields.length === 0) {
          showError('Please add at least one field to scrape');
          return;
        }

        document.getElementById('loading').classList.add('show');
        document.getElementById('loading-text').textContent = 'AI is analyzing the page...';
        document.getElementById('error').classList.remove('show');

        try {
          const response = await fetch('/analyze-page', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ url, fields })
          });

          const data = await response.json();

          if (data.success) {
            aiSuggestedSelectors = data.suggestions;
            showAISuggestions(data.suggestions);
          } else {
            showError(data.error);
          }
        } catch (error) {
          showError('AI analysis failed: ' + error.message);
        } finally {
          document.getElementById('loading').classList.remove('show');
        }
      }

      function showAISuggestions(suggestions) {
        const resultsContent = document.getElementById('resultsContent');
        resultsContent.innerHTML = `
          <div class="info-msg">
            <strong>üß† AI Found These Selectors:</strong><br><br>
            <strong>Container:</strong> <code>${suggestions.container}</code><br><br>
            <strong>Fields:</strong>
            <pre style="margin-top: 10px; background: white; padding: 10px; border-radius: 5px;">${JSON.stringify(suggestions.selectors, null, 2)}</pre>
            <button class="btn btn-success" onclick="useSuggestedSelectors()" style="margin-top: 15px">
              ‚úì Use These Selectors
            </button>
          </div>
        `;
        document.getElementById('results').classList.add('show');
      }

      function useSuggestedSelectors() {
        if (!aiSuggestedSelectors) return;

        // Fill in the manual mode with AI suggestions
        document.getElementById('container').value = aiSuggestedSelectors.container;
        
        const selectorsContainer = document.getElementById('selectors');
        selectorsContainer.innerHTML = '';

        for (const [key, value] of Object.entries(aiSuggestedSelectors.selectors)) {
          const row = document.createElement('div');
          row.className = 'selector-row';
          row.innerHTML = `
            <div><input type="text" value="${key}" class="selector-key"></div>
            <div><input type="text" value="${value}" class="selector-value"></div>
            <div><input type="text" placeholder="Attribute (optional)" class="selector-attribute"></div>
            <div><button class="btn btn-danger btn-small" onclick="removeSelector(this)">‚úï</button></div>
          `;
          selectorsContainer.appendChild(row);
        }

        // Auto-trigger scrape
        scrapeData();
      }

      async function scrapeData() {
        const url = document.getElementById('url').value;
        let container, selectors;

        if (currentMode === 'ai' && aiSuggestedSelectors) {
          container = aiSuggestedSelectors.container;
          selectors = Object.entries(aiSuggestedSelectors.selectors).map(([key, value]) => ({
            key,
            value,
            attribute: null
          }));
        } else {
          container = document.getElementById('container').value;
          const selectorRows = document.querySelectorAll('.selector-row');
          selectors = [];

          selectorRows.forEach(row => {
            const key = row.querySelector('.selector-key').value;
            const value = row.querySelector('.selector-value').value;
            const attribute = row.querySelector('.selector-attribute').value;

            if (key && value) {
              selectors.push({ key, value, attribute: attribute || null });
            }
          });
        }

        if (!url || !container || selectors.length === 0) {
          showError('Please fill in all required fields');
          return;
        }

        document.getElementById('loading').classList.add('show');
        document.getElementById('loading-text').textContent = 'Scraping data...';
        document.getElementById('error').classList.remove('show');
        document.getElementById('results').classList.remove('show');

        try {
          const response = await fetch('/scrape', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ url, container, selectors })
          });

          const data = await response.json();

          if (data.success) {
            scrapedData = data;
            displayResults(data);
          } else {
            showError(data.error);
          }
        } catch (error) {
          showError('Failed to scrape: ' + error.message);
        } finally {
          document.getElementById('loading').classList.remove('show');
        }
      }

      function displayResults(data) {
        const resultsContent = document.getElementById('resultsContent');
        resultsContent.innerHTML = `
          <div class="success-msg">
            ‚úÖ Successfully scraped ${data.count} items from ${data.url}
          </div>
        `;

        data.data.forEach((item, index) => {
          const itemDiv = document.createElement('div');
          itemDiv.className = 'result-item';
          itemDiv.innerHTML = `
            <strong>Item ${index + 1}</strong>
            <pre style="margin-top: 10px; background: #f8f9fa; padding: 10px; border-radius: 5px; overflow-x: auto;">${JSON.stringify(item, null, 2)}</pre>
          `;
          resultsContent.appendChild(itemDiv);
        });

        document.getElementById('results').classList.add('show');
      }

      function showError(message) {
        const errorDiv = document.getElementById('error');
        errorDiv.textContent = '‚ùå ' + message;
        errorDiv.classList.add('show');
      }

      function downloadData() {
        if (!scrapedData) return;

        const dataStr = JSON.stringify(scrapedData, null, 2);
        const blob = new Blob([dataStr], { type: 'application/json' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `scraped_data_${new Date().getTime()}.json`;
        a.click();
        URL.revokeObjectURL(url);
      }
    </script>
